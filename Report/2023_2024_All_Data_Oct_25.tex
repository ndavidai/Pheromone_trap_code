\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{Opening line}: starts with three backticks +
  \texttt{\{r\ ...\}}

  \begin{itemize}
  \tightlist
  \item
    Contains the chunk name (\texttt{chunk\_name}) and options
    (\texttt{echo}, \texttt{eval}, etc.)\\
  \end{itemize}
\item
  \textbf{Code}: written below the opening line\\
\item
  \textbf{Closing line}: three backticks alone
  (\texttt{\textbackslash{}}```)

  \begin{itemize}
  \tightlist
  \item
    \textbf{Nothing else should be on this line}\\
  \end{itemize}
\item
  Optional: you can add comments inside the chunk (\texttt{\#}) to
  describe what the code does
\end{enumerate}
